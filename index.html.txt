        #ui { position: absolute; top: 20px; left: 20px; color: white; pointer-events: none; text-shadow: 0 0 10px rgba(0,255,255,0.5); }
        #level-badge { background: #00f2fe; color: black; padding: 2px 10px; border-radius: 4px; font-weight: bold; margin-right: 10px; text-transform: uppercase; transition: all 0.5s; }
        #score { font-size: 24px; font-weight: bold; }
        
        .overlay { position: fixed; inset: 0; background: rgba(0,0,0,0.85); display: flex; flex-direction: column; align-items: center; justify-content: center; color: white; z-index: 10; text-align: center; }
        h1 { font-size: 48px; margin-bottom: 10px; letter-spacing: 5px; background: linear-gradient(to right, #00f2fe, #4facfe); -webkit-background-clip: text; -webkit-text-fill-color: transparent; }
        button { padding: 15px 40px; font-size: 18px; cursor: pointer; background: transparent; border: 2px solid #00f2fe; color: #00f2fe; border-radius: 30px; transition: 0.3s; margin-top: 20px; text-transform: uppercase; letter-spacing: 2px; }

    <div id="ui">
        <span id="level-badge">Easy</span>

    <div id="start-menu" class="overlay">
        <h1>NEON HIGHWAY</h1>
        <p>Use LEFT/RIGHT arrows to steer. Avoid the traffic.</p>
        <button onclick="startGame()">Start Engine</button>
    </div>

    <div id="game-over" class="overlay">
        <h1 style="background: linear-gradient(to right, #ff0844, #ffb199); -webkit-background-clip: text;">WRECKED</h1>
        <p id="final-score">Score: 0</p>
        <button onclick="resetGame()">Try Again</button>
    </div>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script>
        let scene, camera, renderer, clock;
        let playerCar, road, roadLines = [];
        let trafficCars = [];
        let gameStatus = 'menu';
        let score = 0;
        let speed = 0.6;
        let targetX = 0;
        const laneWidth = 3.5;

        // Difficulty Settings
        const levels = [
            { name: 'Easy', minScore: 0, speed: 0.7, aggression: 0, color: 0x222222 },
            { name: 'Medium', minScore: 1000, speed: 1.0, aggression: 0.005, color: 0x1a2a6c },
            { name: 'Hard', minScore: 3000, speed: 1.4, aggression: 0.015, color: 0xb21f1f },
            { name: 'Extreme', minScore: 6000, speed: 1.8, aggression: 0.03, color: 0x4b0082 },
            { name: 'Impossible', minScore: 10000, speed: 2.3, aggression: 0.06, color: 0x000000 }
        ];
        let currentLevelIndex = 0;

        init();

        function init() {
            scene = new THREE.Scene();
            scene.fog = new THREE.Fog(0x000000, 20, 150);
            
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(0, 4, 12);
            camera.lookAt(0, 0, -5);

            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setClearColor(0x000000);
            document.body.appendChild(renderer.domElement);

            clock = new THREE.Clock();

            // Lighting
            const ambient = new THREE.AmbientLight(0xffffff, 0.4);
            scene.add(ambient);
            const sun = new THREE.DirectionalLight(0xffffff, 1);
            sun.position.set(5, 10, 5);
            scene.add(sun);

            // Road
            const roadGeo = new THREE.PlaneGeometry(15, 2000);
            const roadMat = new THREE.MeshStandardMaterial({ color: 0x222222 });
            road = new THREE.Mesh(roadGeo, roadMat);
            road.rotation.x = -Math.PI / 2;
            scene.add(road);

            // Center Lines
            for(let i=0; i<40; i++) {
                const lineGeo = new THREE.PlaneGeometry(0.3, 4);
                const lineMat = new THREE.MeshBasicMaterial({ color: 0xffffff });
                const line = new THREE.Mesh(lineGeo, lineMat);
                line.rotation.x = -Math.PI / 2;
                line.position.set(0, 0.01, -i * 15);
                roadLines.push(line);
                scene.add(line);
            }

            // Player Car
            const body = new THREE.BoxGeometry(1.8, 0.8, 4);
            const mat = new THREE.MeshStandardMaterial({ color: 0xff0044, metalness: 0.8, roughness: 0.2 });
            playerCar = new THREE.Mesh(body, mat);
            playerCar.position.y = 0.5;
            scene.add(playerCar);

            // Traffic
            spawnTraffic();

            window.addEventListener('keydown', (e) => {
                if(gameStatus !== 'playing') return;
                if(e.key === "ArrowLeft") targetX = Math.max(targetX - laneWidth, -laneWidth);
                if(e.key === "ArrowRight") targetX = Math.min(targetX + laneWidth, laneWidth);
            });

            window.addEventListener('resize', () => {
                camera.aspect = window.innerWidth / window.innerHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(window.innerWidth, window.innerHeight);
            });

            animate();
        }

        function spawnTraffic() {
            trafficCars.forEach(c => scene.remove(c));
            trafficCars = [];
            for (let i = 0; i < 8; i++) {
                const aiCar = new THREE.Mesh(
                    new THREE.BoxGeometry(1.8, 1, 4),
                    new THREE.MeshStandardMaterial({ color: 0x00f2fe })
                );
                resetAiCar(aiCar);
                aiCar.position.z = -50 - (i * 30);
                trafficCars.push(aiCar);
                scene.add(aiCar);
            }
        }

        function resetAiCar(car) {
            car.position.z = -200;
            car.targetX = (Math.floor(Math.random() * 3) - 1) * laneWidth;
            car.position.x = car.targetX;
            car.isSwitching = false;
        }

        function startGame() {
            document.getElementById('start-menu').classList.add('hidden');
            gameStatus = 'playing';
        }

        function resetGame() {
            score = 0;
            currentLevelIndex = 0;
            speed = levels[0].speed;
            playerCar.position.x = 0;
            targetX = 0;
            road.material.color.setHex(levels[0].color);
            document.getElementById('level-badge').innerText = levels[0].name;
            document.getElementById('game-over').style.display = 'none';
            spawnTraffic();
            gameStatus = 'playing';
        }

        function updateDifficulty() {
            const nextLevel = levels[currentLevelIndex + 1];
            if (nextLevel && score >= nextLevel.minScore) {
                currentLevelIndex++;
                speed = levels[currentLevelIndex].speed;
                road.material.color.setHex(levels[currentLevelIndex].color);
                
                const badge = document.getElementById('level-badge');
                badge.innerText = levels[currentLevelIndex].name;
                badge.style.background = (currentLevelIndex >= 3) ? '#ff0844' : '#00f2fe';
            }
        }

        function animate() {
            requestAnimationFrame(animate);
            if(gameStatus !== 'playing') return;

            const delta = clock.getDelta();
            const aggression = levels[currentLevelIndex].aggression;

            // Move Road Lines
            roadLines.forEach(line => {
                line.position.z += speed * 50 * delta;
                if(line.position.z > 20) line.position.z -= 400;
            });

            // Smooth Player Steering
            playerCar.position.x += (targetX - playerCar.position.x) * 0.15;
            playerCar.rotation.z = (playerCar.position.x - targetX) * 0.1;

            // AI Logic
            trafficCars.forEach(car => {
                car.position.z += (speed * 0.7) * 50 * delta;

                // Aggressive Lane Switching
                if (!car.isSwitching && Math.random() < aggression) {
                    car.isSwitching = true;
                    // In Impossible, they hunt you
                    if(currentLevelIndex === 4) {
                        car.targetX = targetX;
                    } else {
                        car.targetX = (Math.floor(Math.random() * 3) - 1) * laneWidth;
                    }
                }
                car.position.x += (car.targetX - car.position.x) * 0.05;

                // Reset Car
                if(car.position.z > 20) resetAiCar(car);

                // Collision Detection
                const dx = Math.abs(car.position.x - playerCar.position.x);
                const dz = Math.abs(car.position.z - playerCar.position.z);
                if(dx < 1.5 && dz < 3.5) {
                    gameStatus = 'gameover';
                    document.getElementById('game-over').style.display = 'flex';
                    document.getElementById('final-score').innerText = `Final Score: ${Math.floor(score)}`;
                }
            });

            score += speed;
            document.getElementById('score').innerText = Math.floor(score).toString().padStart(6, '0');
            updateDifficulty();

            renderer.render(scene, camera);
        }
    </script>
